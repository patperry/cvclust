\section{CV error with two multivariate normal distributed clusters }
\subsection{Setup}
There are two clusters $G_1$ and $G_2$, where observations from $G_1$ are distributed as
\[	N\left( \begin{pmatrix} 
    \mu^X_1 \\
    \mu^Y_1 \\
  \end{pmatrix}, \mathbf{I} \right)	\]
and observations from $G_2$ are distributed as
\[	N\left( \begin{pmatrix} 
    -\mu^X_1 \\
    -\mu^Y_1 \\
  \end{pmatrix}, \mathbf{I} \right)	\]
where $\mu^X_1 > 0$ and $\mu^Y_1 > 0$. If the true cluster is single cluster $G$ with 
\[	P(G=G_1) = P(G=G_2) = 1/2	\]
After applying $K$-means on $Y$ axis with $k=2$ to the whole data, and assume the observation number $n \to \infty$, we have the estimated center of $G_1$ be  
\begin{equation}
\bar{\mu}^Y_1 = 2\varphi(\mu^Y_1)+2\mu^Y_1\Phi(\mu^Y_1)-\mu^Y_1 
\end{equation}
and the estimated center of $G_2$ be  
\begin{equation}
 \bar{\mu}^Y_2 = -2\varphi(\mu^Y_1)-2\mu^Y_1\Phi(\mu^Y_1)+\mu^Y_1 
 \end{equation}
 where $\varphi()$ and $\Phi()$ are the standard normal probability and cumulative distribution function respectively.  
\subsection{CV error with $k=1$ and $k=2$}
By symmetry, the CV error for points from $G_1$ is same as the points from $G_2$. Because $P(G=G_1) = P(G=G_2) =1/2$, 
the CV error for $k=2$ can be calculated solely from $G_2$, that is
\begin{align}
	 CV(2)\hspace{0.05 in}&= E[(Y - \hat{Y})^2], \hspace{0.2in}Y\sim N(-\mu^Y_1,1)   \nonumber \\ \nonumber
	 &= P(\hat{G}=2 | G =2)\cdot E[(Y - \hat{Y})^2 | \hat{G}=2]+P(\hat{G}=1 | G =2)\cdot E[(Y - \hat{Y})^2 | \hat{G}=1] \\ \nonumber
	 &= P(\hat{G}=2 | G =2)\cdot E[(Y - \bar{\mu}^Y_2)^2]+P(\hat{G}=1 | G =2)\cdot E[(Y - \bar{\mu}^Y_1)^2]   \\ \nonumber
	 &= \Phi(\mu^X_1)[var(Y)+(-\mu^Y_1 - \bar{\mu}^Y_2)^2]+ [1-\Phi(\mu^X_1)][var(Y)+(-\mu^Y_1 - \bar{\mu}^Y_1)^2]  \\ \nonumber
	 &= \Phi(\mu^X_1)[1+(\mu^Y_1 + \bar{\mu}^Y_2)^2]+ [1-\Phi(\mu^X_1)][1+(\mu^Y_1 + \bar{\mu}^Y_1)^2]    \\ \nonumber
\bar{\mu}^Y_2 = -\bar{\mu}^Y_1 \hspace{0.05in}	  &= 1+(\mu^Y_1 + \bar{\mu}^Y_1)^2 -4\Phi(\mu^X_1)\mu^Y_1\bar{\mu}^Y_1  \\ \nonumber
\end{align}
where $\bar{\mu}^Y_1$ is given in equation $(1)$.\\\\
When $k=1$, the result is straight forward since the estimated center will be $(0,0)$, so
\[  CV(1) = E[Y^2] = 1 + (\mu^Y_1)^2 \]
where $Y \sim N(-\mu^Y_1,1)$ or  $Y \sim N(\mu^Y_1,1)$
